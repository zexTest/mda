import pandas as pd, numpy as np, matplotlib.pyplot as plt
from sklearn.linear_model import LinearRegression
from sklearn.ensemble import RandomForestRegressor
from sklearn.metrics import mean_absolute_error, mean_squared_error
from pmdarima import auto_arima

df = pd.read_csv('sales_dataset.csv', parse_dates=['Month'], index_col='Month')
df['Time'] = np.arange(len(df))
train, test = df.iloc[:-3], df.iloc[-3:]

models = {
    "LR": LinearRegression(),
    "RF": RandomForestRegressor(100, random_state=0)
}

preds = {}

for name, model in models.items():
    model.fit(train[['Time']], train['Sales'])
    preds[name] = model.predict(test[['Time']])

arima = auto_arima(train['Sales'])
preds["ARIMA"] = arima.predict(n_periods=3)

results = test.copy()
for k,v in preds.items(): results[k] = v

eval_df = pd.DataFrame({
    k: [mean_absolute_error(test['Sales'], v),
        np.sqrt(mean_squared_error(test['Sales'], v))]
    for k,v in preds.items()
}, index=['MAE','RMSE']).T

print("\nModel Performance:\n", eval_df)

df['Sales'].plot(marker='o', label='Actual')
results.drop(columns=['Sales']).plot(marker='x')
plt.legend(); plt.show()

print("\nPredictions:\n", results)

import nltk
from nltk.tokenize import sent_tokenize, word_tokenize, RegexpTokenizer, TweetTokenizer
from nltk.stem import PorterStemmer, WordNetLemmatizer
from nltk.corpus import wordnet

nltk.download('punkt'); nltk.download('wordnet'); nltk.download('averaged_perceptron_tagger')

text = input("Enter Text:\n")

# Tokenization
sents = sent_tokenize(text)
words = word_tokenize(text)

print("\nSentences:", sents)
print("\nWords:", words)
print("\nRegex:", RegexpTokenizer(r'\w+').tokenize(text))
print("\nTweet:", TweetTokenizer().tokenize(text))

# Stemming & Lemmatization
stem = PorterStemmer().stem
lemma = WordNetLemmatizer().lemmatize

print("\nStemming:", [stem(w) for w in words])
print("\nLemmatization:", [lemma(w) for w in words])

# POS-based Lemmatization (optimized)
pos_tags = nltk.pos_tag(words)
pos_map = {"J": wordnet.ADJ, "N": wordnet.NOUN, "V": wordnet.VERB, "R": wordnet.ADV}

lem_pos = [lemma(w, pos_map.get(t[0], wordnet.NOUN)) for w, t in pos_tags]
print("\nLemmatization with POS:", lem_pos)
