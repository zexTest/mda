ARIMA IN PYTHON
import pandas as pd
from statsmodels.tsa.arima.model import ARIMA
import matplotlib.pyplot as plt

# load data
df = pd.read_csv(input())
df.columns = ["Date", "Value"]
df["Date"] = pd.to_datetime(df["Date"])
ts = df.set_index("Date")["Value"]

# model
model = ARIMA(ts, order=(1,1,1)).fit()

# forecast
f = model.forecast(10)
print(f)

# plot
plt.plot(ts)
plt.plot(f)
plt.show()

LINEAR REGRESSION
import pandas as pd
import numpy as np
from sklearn.linear_model import LinearRegression
from sklearn.metrics import mean_squared_error

# load
df = pd.read_csv("data.csv")
y = df.iloc[:,1]

# split
n = int(0.8 * len(y))
x_train = np.arange(n).reshape(-1,1)
x_test = np.arange(n, len(y)).reshape(-1,1)

y_train = y[:n]
y_test = y[n:]

# model
model = LinearRegression().fit(x_train, y_train)
pred = model.predict(x_test)

# evaluation
print("MSE:", mean_squared_error(y_test, pred))

CLASSIFYING SMS BASED ON SPAM AND HAM
import pandas as pd
from sklearn.model_selection import train_test_split
from sklearn.feature_extraction.text import CountVectorizer
from sklearn.linear_model import LogisticRegression

# load data
df = pd.read_csv("spam.csv", encoding="latin-1")[["v1","v2"]]
df.columns = ["label","text"]

# split
X_train, X_test, y_train, y_test = train_test_split(df["text"], df["label"], test_size=0.3)

# convert text → numbers
cv = CountVectorizer()
X_train = cv.fit_transform(X_train)
X_test = cv.transform(X_test)

# model
model = LogisticRegression().fit(X_train, y_train)

# prediction
pred = model.predict(X_test)

print(pred)







ARTIFICIAL NEURAL NETWORK
import pandas as pd
from sklearn.model_selection import train_test_split
from sklearn.preprocessing import normalize
from keras.models import Sequential
from keras.layers import Dense

# load
df = pd.read_csv("data.csv")
X = df.iloc[:,:-1]
y = df.iloc[:,-1]

# split
X_train, X_test, y_train, y_test = train_test_split(X, y, test_size=0.3)

# normalize
X_train = normalize(X_train)
X_test = normalize(X_test)

# model
model = Sequential()
model.add(Dense(10, input_dim=X_train.shape[1], activation='relu'))
model.add(Dense(1, activation='sigmoid'))

model.compile(loss='binary_crossentropy', optimizer='adam', metrics=['accuracy'])

# train
model.fit(X_train, y_train, epochs=10)

# evaluate
print(model.evaluate(X_test, y_test))





ANN IMPROVED ACCURACY
import pandas as pd
from sklearn.model_selection import train_test_split
from sklearn.preprocessing import StandardScaler
from keras.models import Sequential
from keras.layers import Dense

# load
df = pd.read_csv("data.csv")

# convert target
df['chd'] = df['chd'].map({'Si':1,'No':0})

X = df.iloc[:,:-1]
y = df['chd']

# split
X_train, X_test, y_train, y_test = train_test_split(X, y, test_size=0.3)

# scale
sc = StandardScaler()
X_train = sc.fit_transform(X_train)
X_test = sc.transform(X_test)

# model
model = Sequential()
model.add(Dense(10, input_dim=X_train.shape[1], activation='relu'))
model.add(Dense(1, activation='sigmoid'))

model.compile(optimizer='adam', loss='binary_crossentropy', metrics=['accuracy'])

# train
model.fit(X_train, y_train, epochs=20, batch_size=8)

# test
print(model.evaluate(X_test, y_test))

# Load dataset
weatherdata <- read.csv("C:\\Users\\student\\Downloads\\weather (1).csv")

# Clean text
tweets <- weatherdata$tweet_text %>%
  iconv(to = "utf-8") %>%
  tolower()

# Tokenize + remove stopwords
tidy_words <- tibble(text = tweets) %>%
  unnest_tokens(word, text) %>%
  anti_join(stop_words)

# Word frequency
word_freq <- tidy_words %>%
  count(word, sort = TRUE)

# Wordcloud
set.seed(222)
wordcloud(words = word_freq$word,
          freq = word_freq$n,
          max.words = 150,
          min.freq = 5,
          colors = brewer.pal(8, "Dark2"))

# Sentiment analysis (NRC)

# Plot sentiment

        las = 2,
        col = rainbow(10),
        ylab = "count",
